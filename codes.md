SELECT itemid, label, valueuom, count
FROM mimiciii.d_labitems
  RIGHT JOIN (
		SELECT itemid, valueuom, COUNT(*) AS count
		FROM mimiciii.labevents
		WHERE itemid IN (
			50868,
			50862,
			51144,
			50882,
			50885,
			50912,
			50902,

			51274,
			50983,
			50824,

		)

  ) AS le USING (itemid)
ORDER BY label, valueuom
